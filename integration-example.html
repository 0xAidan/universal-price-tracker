‚ùå Old Way (Limited Storage)
// Old storage system - basic localStorage
let historicalData = {};

function addDataPoint(symbol, price) {
    if (!historicalData[symbol]) {
        historicalData[symbol] = [];
    }
    
    historicalData[symbol].push({
        timestamp: Date.now(),
        price: price
    });

    // Simple size limit
    if (historicalData[symbol].length > 1000) {
        historicalData[symbol] = 
            historicalData[symbol].slice(-500);
    }
    
    // Save everything as one big JSON
    localStorage.setItem('priceData', 
        JSON.stringify(historicalData));
}

function getHistoricalData(symbol, period) {
    const data = historicalData[symbol] || [];
    const now = Date.now();
    let startTime;
    
    switch (period) {
        case '1W':
            startTime = now - (7 * 24 * 60 * 60 * 1000);
            break;
        case '1M':
            startTime = now - (30 * 24 * 60 * 60 * 1000);
            break;
        // Limited time periods...
    }
    
    return data.filter(point => 
        point.timestamp >= startTime);
}
‚úÖ New Way (Enhanced Storage)
// Enhanced storage system - multi-layer
const enhancedStorage = new EnhancedDataStorage();

function addDataPoint(symbol, price) {
    // That's it! The enhanced system handles:
    // - Automatic compression
    // - Multiple storage layers
    // - Size management
    // - Performance optimization
    enhancedStorage.addDataPoint(symbol, price);
}

function getHistoricalData(symbol, period) {
    // Works with any time period:
    // '1H', '24H', '1W', '1M', '3M', '1Y', 'ALL'
    return enhancedStorage.getData(symbol, period);
}

// New capabilities you get for free:
function getStorageStats() {
    return enhancedStorage.getStorageStats();
}

function compressOldData() {
    enhancedStorage.compressData();
}

function exportAllData() {
    enhancedStorage.exportData();
}

Step 2:
Replace 2 functions in your existing code:

üîß Exact Migration Code
// 1. Add this at the top of your existing price tracker
const enhancedStorage = new EnhancedDataStorage();

// 2. Find this function in your code:
function updateItemPrice(item) {
    // ... your existing price calculation ...
    
    // OLD: addDataPoint(item.symbol, newPrice);
    // NEW: 
    enhancedStorage.addDataPoint(item.symbol, newPrice);
    
    // Everything else stays the same!
}

// 3. Find your chart rendering code:
function renderChart(symbol, period) {
    // OLD: const data = getHistoricalData(symbol, period);
    // NEW:
    const data = enhancedStorage.getData(symbol, period);
    
    // Your existing chart code works unchanged!
    return data.map(point => ({
        x: point.timestamp,
        y: point.price
    }));
}

// 5. Add historical data import (optional):
function importScrapedData(scrapedData) {
    Object.entries(scrapedData.data).forEach(([source, sourceData]) => {
        Object.entries(sourceData).forEach(([symbol, dataPoints]) => {
            dataPoints.forEach(point => {
                enhancedStorage.addDataPoint(symbol, point.price, point.timestamp);
            });
        });
    });
